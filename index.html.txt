<body>
  <h1>Game Collection Tracker</h1>

  <div class="topbar">
    <input type="text" id="searchInput" placeholder="Search by game name..." />
    <button onclick="filterGames()">Search</button>
    <button onclick="loadGames()">Reset</button>
  </div>

  <div class="form-section">
    <h2>Add New Game</h2>
    <div class="form-grid">
      <input type="text" id="gameName" placeholder="Game Name" />
      <input type="text" id="genre" placeholder="Genre" />
      <input type="number" id="developerId" placeholder="Developer ID" />
    </div>
    <div style="margin-top: 12px;">
      <button onclick="addGame()">Add Game</button>
    </div>
    <div class="message" id="message"></div>
  </div>

  <div class="details-section">
    <h2>Game Details</h2>
    <div class="details-box" id="detailsBox">Click "View Details" on any game card.</div>
  </div>

  <div class="games" id="games"></div>

  <script>
    let allGames = [];

    async function loadGames() {
      const res = await fetch("http://localhost:3000/api/games");
      const result = await res.json();

      const gamesDiv = document.getElementById("games");
      gamesDiv.innerHTML = "";

      if (!result.success) {
        gamesDiv.innerHTML = "<p>Failed to load games.</p>";
        return;
      }

      allGames = result.data;
      renderGames(allGames);
    }

    function renderGames(games) {
      const gamesDiv = document.getElementById("games");
      gamesDiv.innerHTML = "";

      if (games.length === 0) {
        gamesDiv.innerHTML = "<p>No games found.</p>";
        return;
      }

      games.forEach(game => {
        const card = document.createElement("div");
        card.className = "card";
        card.innerHTML = `
          <h2>${game.game_name}</h2>
          <p><strong>Genre:</strong> ${game.genre}</p>
          <p><strong>Developer:</strong> ${game.developer || "N/A"}</p>
          <p><strong>Game ID:</strong> ${game.game_id}</p>
          <div class="button-row">
            <button onclick="viewDetails(${game.game_id})">View Details</button>
            <button onclick="deleteGame(${game.game_id})">Delete</button>
          </div>
        `;
        gamesDiv.appendChild(card);
      });
    }

    function filterGames() {
      const query = document.getElementById("searchInput").value.toLowerCase().trim();
      const filtered = allGames.filter(game =>
        game.game_name.toLowerCase().includes(query)
      );
      renderGames(filtered);
    }

    async function addGame() {
      const game_name = document.getElementById("gameName").value.trim();
      const genre = document.getElementById("genre").value.trim();
      const developer_id = document.getElementById("developerId").value.trim();
      const message = document.getElementById("message");

      if (!game_name || !genre || !developer_id) {
        message.textContent = "Please fill all fields.";
        return;
      }

      const res = await fetch("http://localhost:3000/api/games", {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify({
          game_name,
          genre,
          developer_id: Number(developer_id)
        })
      });

      const result = await res.json();

      if (result.success) {
        message.textContent = "Game added successfully.";
        document.getElementById("gameName").value = "";
        document.getElementById("genre").value = "";
        document.getElementById("developerId").value = "";
        loadGames();
      } else {
        message.textContent = result.error || "Failed to add game.";
      }
    }

    async function viewDetails(gameId) {
      const res = await fetch(`http://localhost:3000/api/games/${gameId}`);
      const result = await res.json();

      const detailsBox = document.getElementById("detailsBox");

      if (!result.success) {
        detailsBox.textContent = "Could not load game details.";
        return;
      }

      const game = result.data;

      const platformText = game.platforms.length
        ? game.platforms.map(p => p.platform_name).join(", ")
        : "None";

      const priceText = game.prices.length
        ? game.prices.map(p => `$${Number(p.amount).toFixed(2)}`).join(", ")
        : "None";

      const historyText = game.play_history.length
        ? game.play_history.map(p => `${p.player_name}: ${p.hours_played} hrs`).join("\n")
        : "No play history";

      detailsBox.textContent =
`Game Name: ${game.game_name}
Genre: ${game.genre}
Developer: ${game.developer || "N/A"}
Developer ID: ${game.developer_id ?? "N/A"}
Game ID: ${game.game_id}

Platforms: ${platformText}
Prices: ${priceText}

Play History:
${historyText}`;
    }

    async function deleteGame(gameId) {
      const ok = confirm("Are you sure you want to delete this game?");
      if (!ok) return;

      const res = await fetch(`http://localhost:3000/api/games/${gameId}`, {
        method: "DELETE"
      });

      const result = await res.json();

      if (result.success) {
        document.getElementById("detailsBox").textContent = "Game deleted successfully.";
        loadGames();
      } else {
        alert(result.error || "Delete failed.");
      }
    }

    loadGames();
  </script>
